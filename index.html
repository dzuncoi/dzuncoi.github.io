# Send me a message
function send_message() {
  local subject="$1"
  local body="$2"
  echo "Sending: $subject"
  mail -s "$subject" $EMAIL <<< "$body"
  echo "Message sent! I'll get back to you ASAP."
}

# Usage: ./contact.sh "Hello!" "Let's collaborate"
send_message "$1" "$2"
